Infinity Wear - نظام تسجيل المستوردين

تعليمات الاستخدام
  1. افتح صفحة النموذج: https://infinitywearsa.com/importers/register
  2. افتح Console: اضغط F12 ثم اختر Console
  3. انسخ الكود من المربع أدناه والصقه في Console
  4. اضغط Enter لبدء الاختبار
كود الاختبار البسيط
// اختبار بسيط للنموذج - Infinity Wear console.log('🚀 بدء الاختبار البسيط...'); // اختبار 1: تحميل الصفحة console.log('✅ تحميل الصفحة: نجح'); // اختبار 2: هيكل النموذج const requiredElements = [ 'registrationForm', 'quantity', 'design_option', 'business_type', 'company_name', 'city', 'name', 'email', 'phone' ]; const missingElements = requiredElements.filter(id => !document.getElementById(id)); if (missingElements.length === 0) { console.log('✅ هيكل النموذج: نجح - جميع العناصر موجودة'); } else { console.log('❌ هيكل النموذج: فشل - عناصر مفقودة:', missingElements); } // اختبار 3: خيارات التصميم const designOptions = document.querySelectorAll('input[name="design_option"]'); console.log('✅ خيارات التصميم: نجح -', designOptions.length, 'خيارات موجودة'); // اختبار 4: حقل الكمية const quantityInput = document.getElementById('quantity'); if (quantityInput) { quantityInput.value = '100'; console.log('✅ حقل الكمية: نجح - يمكن إدخال القيم'); } else { console.log('❌ حقل الكمية: فشل - الحقل غير موجود'); } // اختبار 5: نوع النشاط const businessTypeSelect = document.getElementById('business_type'); if (businessTypeSelect) { const options = businessTypeSelect.querySelectorAll('option'); console.log('✅ نوع النشاط: نجح -', options.length - 1, 'خيارات متاحة'); } else { console.log('❌ نوع النشاط: فشل - القائمة غير موجودة'); } // اختبار 6: معلومات الشركة const companyName = document.getElementById('company_name'); const city = document.getElementById('city'); if (companyName && city) { companyName.value = 'شركة اختبار'; city.value = 'الرياض'; console.log('✅ معلومات الشركة: نجح - يمكن إدخال البيانات'); } else { console.log('❌ معلومات الشركة: فشل - الحقول غير موجودة'); } // اختبار 7: المعلومات الشخصية const name = document.getElementById('name'); const email = document.getElementById('email'); const phone = document.getElementById('phone'); if (name && email && phone) { name.value = 'أحمد محمد'; email.value = 'test@example.com'; phone.value = '0501234567'; console.log('✅ المعلومات الشخصية: نجح - يمكن إدخال البيانات'); } else { console.log('❌ المعلومات الشخصية: فشل - الحقول غير موجودة'); } // اختبار 8: واجهة التصميم المخصص const customOption = document.querySelector('input[value="custom"]'); if (customOption) { customOption.checked = true; customOption.dispatchEvent(new Event('change', { bubbles: true })); console.log('✅ واجهة التصميم المخصص: نجح - تم اختيار الخيار'); } else { console.log('❌ واجهة التصميم المخصص: فشل - خيار التصميم المخصص غير موجود'); } // اختبار 9: الملخص const summaryElements = [ 'summary_quantity', 'summary_design_option', 'summary_company', 'summary_business_type', 'summary_city', 'summary_name', 'summary_email', 'summary_phone' ]; const existingSummaryElements = summaryElements.filter(id => document.getElementById(id)); console.log('✅ الملخص: نجح -', existingSummaryElements.length, 'عنصر موجود'); // اختبار 10: إعادة التعيين if (typeof window.multiStepForm !== 'undefined' && window.multiStepForm.resetForm) { console.log('✅ إعادة التعيين: نجح - الدالة موجودة'); } else { console.log('❌ إعادة التعيين: فشل - الدالة غير موجودة'); } console.log('🎉 انتهى الاختبار البسيط!');
خطوات الاختبار
  1. افتح صفحة النموذج في تبويب جديد
  2. افتح Console (F12)
  3. انسخ الكود من المربع أعلاه
  4. الصق الكود في Console
  5. اضغط Enter
  6. شاهد النتائج في Console
ملاحظات مهمة
  1. تأكد من فتح صفحة النموذج قبل تشغيل الكود
  2. إذا ظهرت أخطاء، تأكد من تحميل الصفحة بالكامل
  3. الكود آمن ولا يؤثر على البيانات الحقيقية
  4. يمكن تشغيل الكود عدة مرات
